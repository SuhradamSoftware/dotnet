drop table Invoice;
drop table InvoiceDetails;
drop table Transactions;
drop table Taxes;
drop table Product;
drop table Customer;

CREATE TABLE [Invoice] (
[InvoiceId] bigint IDENTITY (1,1) NOT NULL,
[InvoiceNo] nvarchar(20) NULL,
[CustomerID] bigint NULL,
[InvoiceDate] datetime NULL,
[InvoiceAmount] decimal(14, 2) NULL
);

CREATE TABLE [InvoiceDetails] (
[DetailId] bigint IDENTITY (1,1) NOT NULL,
[InvoiceId] bigint NULL,
[ProductID] int NULL,
[Quantity] decimal(14, 2) NULL,
[Price] decimal(14, 2) NULL,
[Amount] decimal(14, 2) NULL
);

CREATE TABLE [Transactions] (
[TransactionId] bigint IDENTITY (1,1) NOT NULL,
[CustomerID] bigint NULL,
[InvoiceId] bigint NULL,
[CreaditAmount] decimal(14, 2) NULL,
[DebitAmount] decimal(14, 2) NULL,
[TransactionDate] datetime NULL
);

CREATE TABLE [Taxes] (
  [TaxID] int IDENTITY (1,1) NOT NULL
, [TaxDescription] nvarchar(100) NULL
, [TaxRate] money NULL
);

CREATE TABLE [Product] (
  [ProductID] bigint IDENTITY (1,1) NOT NULL
, [ProductName] nvarchar(200) NOT NULL
, [ProductDescription] nvarchar(250) NULL
);

CREATE TABLE [Customer] (
  [CustomerID] bigint IDENTITY (1,1) NOT NULL
, [CustomerName] nvarchar(150) NULL
, [Address] nvarchar(250) NULL
, [City] nvarchar(100) NULL
, [PhoneNumber] nvarchar(20) NULL
, [extrano] nvarchar(20) NULL
, [email] nvarchar(40) NULL
, [CreatedByID] int NULL
, [CustomerCode] nvarchar(50) NULL
);

CREATE TABLE [Default] (
  [fld_default_name] nvarchar(30) NULL
, [fld_value] nvarchar(20) NULL
);
